# 参数与结果映射

## 这个问题为什么存在？

> 你传入一个 `User` 对象，MyBatis 怎么知道把 `name` 字段放到 SQL 的哪个 `?` 里？
> 查询结果 `ResultSet` 怎么变成一个 `User` 对象？

这就是 **参数映射**（Java → JDBC）和 **结果映射**（JDBC → Java）要做的事。

---

## 它是怎么解决问题的？

### 参数映射（Java → SQL `?`）

**`#{}` 解析过程**：

```
#{propertyName, javaType=..., jdbcType=..., mode=..., numericScale=..., resultMap=..., typeHandler=..., jdbcTypeName=...}

解析入口：GenericTokenParser.parse()
  → 找到 #{ 和 } 之间的内容
  → 用 ParameterMappingTokenHandler 处理
  → 生成 ? 占位符 + ParameterMapping 对象（记录类型处理器等信息）
```

**`ParameterHandler.setParameters()`**（真正设置 PreparedStatement 参数）：

```java
// DefaultParameterHandler.java
public void setParameters(PreparedStatement ps) {
    List<ParameterMapping> parameterMappings = boundSql.getParameterMappings();
    for (int i = 0; i < parameterMappings.size(); i++) {
        ParameterMapping parameterMapping = parameterMappings.get(i);
        Object value;
        String propertyName = parameterMapping.getProperty();
        // 从入参中取对应属性的值
        if (boundSql.hasAdditionalParameter(propertyName)) {
            value = boundSql.getAdditionalParameter(propertyName);
        } else {
            MetaObject metaObject = MetaObject.forObject(parameterObject);
            value = metaObject.getValue(propertyName);  // ← OGNL 取值
        }
        TypeHandler typeHandler = parameterMapping.getTypeHandler();
        // 用 TypeHandler 设置 PreparedStatement 的参数
        typeHandler.setParameter(ps, i + 1, value, jdbcType);
    }
}
```

### 结果映射（ResultSet → Java 对象）

**`ResultSetHandler.handleResultSets()`**（核心方法）：

```java
// DefaultResultSetHandler.java
@Override
public List<Object> handleResultSets(Statement stmt) {
    List<Object> multipleResults = new ArrayList<>();
    int resultSetCount = 0;
    ResultSet rs = stmt.getResultSet();   // 从这里开始处理结果集
    // 处理每个 ResultSet（存储过程可能有多个结果集）
    while (rs != null) {
        ResultMap resultMap = getResultMap(resultSetCount);  // 找到对应的 ResultMap
        List<Object> resultList = handleResultSet(rs, resultMap, ...);
        multipleResults.add(resultList);
        rs = getNextResultSet(stmt);   // 下一个结果集
    }
    return collapseSingleResultList(multipleResults);
}

// 处理单行结果
private Object getRowValue(ResultSet rs, ResultMap resultMap, ...) {
    // 1. 创建返回对象（通过反射调用默认构造方法或工厂方法）
    Object rowValue = createResultObject(rs, resultMap, ...);
    // 2. 如果有嵌套映射，需要延迟加载或立即加载
    // 3. 遍历 ResultMap 的每个 ResultMapping（每个 <result/> 或 <association/>）
    for (ResultMapping resultMapping : resultMap.getResultMappings()) {
        if (resultMapping.isCompositeResult()) {
            // 嵌套结果映射（<association/> <collection/>）
            handleResultSetForNestedResult(rs, resultMapping, ...);
        } else {
            // 普通字段映射：用 TypeHandler 从 ResultSet 读值，设置到对象属性
            TypeHandler<?> typeHandler = resultMapping.getTypeHandler();
            Object value = typeHandler.getResult(rs, columnName);
            metaObject.setValue(resultMapping.getProperty(), value);
        }
    }
    return rowValue;
}
```

## TypeHandler：映射的核心

**TypeHandler 接口**（参数设置 + 结果读取）：

```java
public interface TypeHandler<T> {
    // 设置 PreparedStatement 参数
    void setParameter(PreparedStatement ps, int i, T parameter, JdbcType jdbcType);
    // 从 ResultSet 读取值（按列名）
    T getResult(ResultSet rs, String columnName);
    // 从 ResultSet 读取值（按列索引）
    T getResult(ResultSet rs, int columnIndex);
    // 从 CallableStatement 读取值（存储过程）
    T getResult(CallableStatement cs, int columnIndex);
}

// 内置的 TypeHandler 示例
// IntegerTypeHandler：Java Integer ↔ JDBC INTEGER
// StringTypeHandler：Java String ↔ JDBC VARCHAR
// DateTypeHandler：Java Date ↔ JDBC TIMESTAMP
```

**自定义 TypeHandler**（如：JSON 字段 ↔ Java 对象）：

```java
@MappedTypes({UserDetail.class})   // 指定 Java 类型
@MappedJdbcTypes({JdbcType.VARCHAR})  // 指定 JDBC 类型
public class JsonTypeHandler extends BaseTypeHandler<Object> {
    private final ObjectMapper objectMapper = new ObjectMapper();

    @Override
    public void setParameter(PreparedStatement ps, int i, Object parameter, JdbcType jdbcType) {
        // Java 对象 → JSON 字符串 → 设置到 PreparedStatement
        String json = objectMapper.writeValueAsString(parameter);
        ps.setString(i, json);
    }

    @Override
    public Object getResult(ResultSet rs, String columnName) {
        // 从 ResultSet 读 JSON 字符串 → 解析成 Java 对象
        String json = rs.getString(columnName);
        return objectMapper.readValue(json, Object.class);
    }
}
```

## XML 映射配置方式

### 自动映射（最常用）

```xml
<select id="selectById" resultType="User">
    SELECT id, user_name AS userName, age FROM user WHERE id = #{id}
    <!-- 自动映射规则：
         ResultSet 列名（或别名）= 类属性名（不区分大小写）
         user_name → userName ✅（开启了 mapUnderscoreToCamelCase 时）
         userName → userName ✅ -->
</select>
```

**`mapUnderscoreToCamelCase=true`**（重要配置）：

```xml
<configuration>
    <settings>
        <!-- 下划线转驼峰自动映射 -->
        <setting name="mapUnderscoreToCamelCase" value="true"/>
    </settings>
</configuration>
```

### ResultMap 手动映射

```xml
<resultMap id="userResultMap" type="User">
    <!-- id 标签：主键映射（有特殊优化） -->
    <id column="id" property="id"/>
    <!-- result 标签：普通字段映射 -->
    <result column="user_name" property="userName"/>
    <result column="age" property="age"/>
    <!-- association：一对一嵌套 -->
    <association property="department" javaType="Department">
        <id column="dept_id" property="id"/>
        <result column="dept_name" property="name"/>
    </association>
    <!-- collection：一对多嵌套 -->
    <collection property="roles" ofType="Role">
        <id column="role_id" property="id"/>
        <result column="role_name" property="name"/>
    </collection>
    <!-- constructor：构造函数注入（没有无参构造时用） -->
    <constructor>
        <idArg column="id" javaType="int"/>
        <arg column="user_name" javaType="String"/>
    </constructor>
</resultMap>

<select id="selectById" resultMap="userResultMap">
    SELECT u.id, u.user_name, u.age, d.name AS dept_name
    FROM user u LEFT JOIN department d ON u.dept_id = d.id
    WHERE u.id = #{id}
</select>
```

## 它和相似方案的本质区别是什么？

| 对比维度 | MyBatis 结果映射 | JPA / Hibernate 自动映射 | Spring JDBC Template |
|---------|----------------|----------------------|-------------------|
| 映射方式 | XML 或注解声明 | 注解（@Entity）或 XML | RowMapper 手动实现 |
| 嵌套对象 | 需配置 `<association>` | 自动（@OneToOne 等） | 手动实现 |
| 性能 | 高（可控 SQL） | 中（N+1 问题） | 高（但代码多） |
| 灵活性 | 最高（SQL 全掌控） | 低（自动生成 SQL） | 中（SQL 手写，映射手动） |

**本质区别**：MyBatis 的映射是 **声明式** 的（XML/注解描述映射规则），而非自动的（JPA）。

## 正确使用方式

### 参数传递的几种方式

```java
// 1. 单个参数（基本类型或对象）
User selectById(Integer id);   // XML 中 #{id} 直接取参数值
User selectByUser(User user); // XML 中 #{name}, #{age} 取属性

// 2. 多个参数（需要用 @Param 指定名字）
List<User> selectByAgeAndName(@Param("age") Integer age, @Param("name") String name);
// XML 中：#{age}, #{name}

// 3. 直接用 Map 传参
List<User> selectByMap(Map<String, Object> params);
// XML 中：#{key1}, #{key2}

// 4. 对象 + @Param 混合
List<User> select(User user, @Param("status") String status);
// XML 中：#{name} (User 的属性), #{status}
```

### TypeHandler 注册

```xml
<!-- mybatis-config.xml -->
<typeHandlers>
    <!-- 自动发现（扫描包） -->
    <package name="com.example.typehandler"/>
    <!-- 或单个注册 -->
    <typeHandler handler="com.example.typehandler.JsonTypeHandler"
                javaType="com.example.UserDetail"
                jdbcType="VARCHAR"/>
</typeHandlers>
```

## 边界情况和坑

### 坑1：参数名推断失败

```xml
<!-- Java 接口 -->
User selectById(Integer id);

<!-- XML：如果编译时保留参数名（javac -parameters），可以直接用 #{id} -->
<!-- 如果没有保留参数名，MyBatis 会用 param1, arg0 等名字 → #{id} 找不到！ -->

<!-- 解决：加 @Param -->
User selectById(@Param("id") Integer id);
```

### 坑2：resultType 和 resultMap 不能同时用

```xml
<!-- ❌ 错误 -->
<select id="..." resultType="User" resultMap="userMap">

<!-- ✅ 正确：二选一 -->
<select id="..." resultType="User">        <!-- 自动映射 -->
<select id="..." resultMap="userMap">      <!-- 手动映射 -->
```

### 坑3：列名和属性名不一致

```xml
<!-- 解决方式一：SQL 中起别名 -->
<select id="..." resultType="User">
    SELECT user_name AS userName FROM user   <!-- AS 别名匹配属性名 -->
</select>

<!-- 解决方式二：开启驼峰转换（推荐） -->
<setting name="mapUnderscoreToCamelCase" value="true"/>
<!-- user_name → userName 自动匹配 -->

<!-- 解决方式三：用 resultMap -->
<resultMap>
    <result column="user_name" property="userName"/>
</resultMap>
```

### 坑4：嵌套查询的 N+1 问题

```xml
<!-- ❌ 导致 N+1：先查 10 个 User，每个 User 又触发 1 次查询 Department -->
<resultMap id="userWithDept" type="User">
    <association property="dept"
               select="selectDeptById"   <!-- 嵌套查询 -->
               column="dept_id"/>
</resultMap>

<!-- ✅ 解决一：用嵌套结果（JOIN 一次查询） -->
<resultMap id="userWithDept" type="User">
    <association property="dept" javaType="Department">
        <id column="dept_id" property="id"/>
        <result column="dept_name" property="name"/>
    </association>
</resultMap>
<select id="..." resultMap="userWithDept">
    SELECT u.*, d.name AS dept_name
    FROM user u LEFT JOIN department d ON u.dept_id = d.id
</select>

<!-- ✅ 解决二：开启延迟加载（减少不必要查询） -->
<setting name="lazyLoadingEnabled" value="true"/>
```

## 我的理解

参数映射和结果映射是 MyBatis 「半自动化 ORM」的核心——
- **参数映射**：把 Java 对象属性值，通过 `TypeHandler` 设置到 `PreparedStatement` 的 `?` 里。
- **结果映射**：把 `ResultSet` 的列，通过 `TypeHandler` 读取值，再通过反射设置到 Java 对象的属性里。

**`TypeHandler` 是扩展的关键**：遇到特殊类型（如 JSON 字段、加密字段），继承 `BaseTypeHandler` 就能无缝接入 MyBatis 的映射流程。

**实际建议**：
- 简单查询用 `resultType` + 驼峰转换 ✅
- 复杂对象（嵌套）用 `resultMap` ✅
- 避免 N+1 问题：优先用 JOIN 一次查询，而非嵌套 `select` ⚠️
